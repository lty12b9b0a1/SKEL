var res = escape("abc<>/'", true)
    console.assert(res == "abc&lt;&gt;/&#x27;")
    res = escape("<>", true)
    console.assert(res == "&lt;&gt;")
    res = escape("abc", true)
    console.assert(res == "abc")
    res = escape("abc&", true)
    console.assert(res == "abc&amp;")

    res = unescape("abc&lt;&gt;/&#x27;")
    console.assert(res == "abc<>/'")
    res = unescape("&lt;&gt;")
    console.assert(res == "<>")
    res = unescape("abc")
    console.assert(res == "abc")
    res = unescape("abc&amp;")
    console.assert(res == "abc&")
